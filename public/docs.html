Code samples

require 'json'

headers = {
  'Authorization' => 'API_KEY'
}

result = RestClient.get 'https://hateoas-demo.herokuapp.com/api/v1/api/v1/health',
  params: {
  }, headers: headers

p JSON.parse(result)


import requests
headers = {
  'Authorization': 'API_KEY'
}

r = requests.get('https://hateoas-demo.herokuapp.com/api/v1/api/v1/health', headers = headers)

print(r.json())



const headers = {
  'Authorization':'API_KEY'
};

fetch('https://hateoas-demo.herokuapp.com/api/v1/api/v1/health',
{
  method: 'GET',

  headers: headers
})
.then(function(res) {
    return res.json();
}).then(function(body) {
    console.log(body);
});
